102/* -------------------------------------------------------------------------- */
103/* Public key as (usually) fetched from DNS */
104typedef struct pdkim_pubkey {
105 const uschar * version; /* v= */
106 const uschar *granularity; /* g= */
107
108 const uschar * hashes; /* h= */
109 const uschar * keytype; /* k= */
110 const uschar * srvtype; /* s= */
111 uschar *notes; /* n= */
112
113 blob key; /* p= */
114 int testing; /* t=y */
115 int no_subdomaining; /* t=s */
116} pdkim_pubkey;
117
118/* -------------------------------------------------------------------------- */
119/* Body-hash to be calculated */
120typedef struct pdkim_bodyhash {
121 struct pdkim_bodyhash * next;
122 int hashtype;
123 int canon_method;
124 long bodylength;
125
126 hctx body_hash_ctx;
127 unsigned long signed_body_bytes; /* done so far */
128 int num_buffered_blanklines;
129
130 blob bh; /* completed hash */
131} pdkim_bodyhash;

133/* -------------------------------------------------------------------------- */
134/* Signature as it appears in a DKIM-Signature header */
135typedef struct pdkim_signature {
136 struct pdkim_signature * next;
137
138 /* Bits stored in a DKIM signature header --------------------------- */
139
140 /* (v=) The version, as an integer. Currently, always "1" */
141 int version;
142
143 /* (a=) The signature algorithm. Either PDKIM_ALGO_RSA_SHA256 */
144 int keytype; /* pdkim_keytypes index */
145 int hashtype; /* pdkim_hashes index */
146
147 /* (c=x/) Header canonicalization method. Either PDKIM_CANON_SIMPLE
148 or PDKIM_CANON_RELAXED */
149 int canon_headers;
150
151 /* (c=/x) Body canonicalization method. Either PDKIM_CANON_SIMPLE
152 or PDKIM_CANON_RELAXED */
153 int canon_body;
154
155 /* (q=) Query Method. Currently, only PDKIM_QUERYMETHOD_DNS_TXT
156 is specified */
157 int querymethod;
158
159 /* (s=) The selector string as given in the signature */
160 uschar *selector;
161
162 /* (d=) The domain as given in the signature */
163 uschar *domain;
164
165 /* (i=) The identity as given in the signature */
166 uschar *identity;
167
168 /* (t=) Timestamp of signature creation */
169 unsigned long created;
170
171 /* (x=) Timestamp of expiry of signature */
172 unsigned long expires;
173
174 /* (l=) Amount of hashed body bytes (after canonicalization). Default
175 is -1. Note: a value of 0 means that the body is unsigned! */
176 long bodylength;
177
178 /* (h=) Colon-separated list of header names that are included in the
179 signature */
180 uschar *headernames;
181
182 /* (z=) */
183 uschar *copiedheaders;
184
185 /* (b=) Raw signature data, along with its length in bytes */
186 blob sighash;
187
188 /* (bh=) Raw body hash data, along with its length in bytes */
189 blob bodyhash;
190
191 /* Folded DKIM-Signature: header. Signing only, NULL for verifying.
192 Ready for insertion into the message. Note: Folded using CRLFTB,
193 but final line terminator is NOT included. Note2: This buffer is
194 free()d when you call pdkim_free_ctx(). */
195 uschar *signature_header;
196
197 /* The main verification status. Verification only. One of:
198
199 PDKIM_VERIFY_NONE Verification was not attempted. This status
200 should not appear.
201
202 PDKIM_VERIFY_INVALID There was an error while trying to verify
203 the signature. A more precise description
204 is available in verify_ext_status.
205
206 PDKIM_VERIFY_FAIL Verification failed because either the body
207 hash did not match, or the signature verification
208 failed. This means the message was modified.
209 Check verify_ext_status for the exact reason.
210
211 PDKIM_VERIFY_PASS Verification succeeded.
212 */
213 int verify_status;
214
215 /* Extended verification status. Verification only. Depending on the value
216 of verify_status, it can contain:
217
218 For verify_status == PDKIM_VERIFY_INVALID:
219
220 PDKIM_VERIFY_INVALID_PUBKEY_UNAVAILABLE
221 Unable to retrieve a public key container.
222
223 PDKIM_VERIFY_INVALID_BUFFER_SIZE
224 Either the DNS name constructed to retrieve the public key record
225 does not fit into PDKIM_DNS_TXT_MAX_NAMELEN bytes, or the retrieved
226 record is longer than PDKIM_DNS_TXT_MAX_RECLEN bytes.
227
228 PDKIM_VERIFY_INVALID_PUBKEY_PARSING
229 (Syntax) error while parsing the retrieved public key record.
230
231
232 For verify_status == PDKIM_VERIFY_FAIL:
233
234 PDKIM_VERIFY_FAIL_BODY
235 The calculated body hash does not match the advertised body hash
236 from the bh= tag of the signature.
237
238 PDKIM_VERIFY_FAIL_MESSAGE
239 RSA verification of the signature (b= tag) failed.
240 */
241 int verify_ext_status;
242
243 /* Pointer to a public key record that was used to verify the signature.
244 See pdkim_pubkey declaration above for more information.
245 Caution: is NULL if signing or if no record was retrieved. */
246 pdkim_pubkey *pubkey;
247
248 /* Properties below this point are used internally only ------------- */
249
250 /* Per-signature helper variables ----------------------------------- */
251 pdkim_bodyhash *calc_body_hash; /* hash to be / being calculated */
252
253 pdkim_stringlist *headers; /* Raw headers included in the sig */
254
255 /* Signing specific ------------------------------------------------- */
256 uschar * privkey; /* Private key */
257 uschar * sign_headers; /* To-be-signed header names */
258 uschar * rawsig_no_b_val; /* Original signature header w/o b= tag value. */
259} pdkim_signature;

262/* -------------------------------------------------------------------------- */
263/* Context to keep state between all operations. */
264typedef struct pdkim_ctx {
265
266#define PDKIM_MODE_SIGN BIT(0) /* if unset, mode==verify */
267#define PDKIM_DOT_TERM BIT(1) /* dot termination and unstuffing */
268#define PDKIM_SEEN_CR BIT(2)
269#define PDKIM_SEEN_LF BIT(3)
270#define PDKIM_PAST_HDRS BIT(4)
271#define PDKIM_SEEN_EOD BIT(5)
272 unsigned flags;
273
274 /* One (signing) or several chained (verification) signatures */
275 pdkim_signature *sig;
276
277 /* One (signing) or several chained (verification) bodyhashes */
278 pdkim_bodyhash *bodyhash;
279
280 /* Callback for dns/txt query method (verification only) */
281 uschar * (*dns_txt_callback)(char *);
282
283 /* Coder's little helpers */
284 gstring *cur_header;
285 uschar *linebuf;
286 int linebuf_offset;
287 int num_headers;
288 pdkim_stringlist *headers; /* Raw headers for verification */
289} pdkim_ctx;
