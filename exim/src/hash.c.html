33BOOL
34exim_sha_init(hctx * h, hashmethod m)
35{
36/*XXX extend for sha512 */
37switch (h->method = m)
38 {
39 case HASH_SHA1: h->hashlen = 20; SHA1_Init (&h->u.sha1); break;
40 case HASH_SHA2_256: h->hashlen = 32; SHA256_Init(&h->u.sha2_256); break;
41 case HASH_SHA2_384: h->hashlen = 48; SHA384_Init(&h->u.sha2_512); break;
42 case HASH_SHA2_512: h->hashlen = 64; SHA512_Init(&h->u.sha2_512); break;
43#ifdef EXIM_HAVE_SHA3
44 case HASH_SHA3_224: h->hashlen = 28;
45 EVP_DigestInit(h->u.mctx = EVP_MD_CTX_new(), EVP_sha3_224());
46 break;
47 case HASH_SHA3_256: h->hashlen = 32;
48 EVP_DigestInit(h->u.mctx = EVP_MD_CTX_new(), EVP_sha3_256());
49 break;
50 case HASH_SHA3_384: h->hashlen = 48;
51 EVP_DigestInit(h->u.mctx = EVP_MD_CTX_new(), EVP_sha3_384());
52 break;
53 case HASH_SHA3_512: h->hashlen = 64;
54 EVP_DigestInit(h->u.mctx = EVP_MD_CTX_new(), EVP_sha3_512());
55 break;
56#endif
57 default: h->hashlen = 0; return FALSE;
58 }
59return TRUE;
60}

64exim_sha_update(hctx * h, const uschar * data, int len)
65{
66switch (h->method)
67 {
68 case HASH_SHA1: SHA1_Update (&h->u.sha1, data, len); break;
69 case HASH_SHA2_256: SHA256_Update(&h->u.sha2_256, data, len); break;
70 case HASH_SHA2_384: SHA384_Update(&h->u.sha2_512, data, len); break;
71 case HASH_SHA2_512: SHA512_Update(&h->u.sha2_512, data, len); break;
72#ifdef EXIM_HAVE_SHA3
73 case HASH_SHA3_224:
74 case HASH_SHA3_256:
75 case HASH_SHA3_384:
76 case HASH_SHA3_512: EVP_DigestUpdate(h->u.mctx, data, len); break;
77#endif
78 /* should be blocked by init not handling these, but be explicit to
79 guard against accidents later (and hush up clang -Wswitch) */
80 default: assert(0);
81 }
82}

93 case HASH_SHA2_384: SHA384_Final(b->data, &h->u.sha2_512); break;
94 case HASH_SHA2_512: SHA512_Final(b->data, &h->u.sha2_512); break;
95#ifdef EXIM_HAVE_SHA3

97 case HASH_SHA3_256:
98 case HASH_SHA3_384:
99 case HASH_SHA3_512: EVP_DigestFinal(h->u.mctx, b->data, NULL); break;
100#endif
101 default: assert(0);
102 }

242static void
243native_sha1_start(sha1 *base)
244{
245base->H[0] = 0x67452301;
246base->H[1] = 0xefcdab89;

255/*************************************************
256* Process another 64-byte block *
257*************************************************/
258
259/* This function implements central part of the algorithm
260
261Arguments:
262 base pointer to sha1 storage structure
263 text pointer to next 64 bytes of subject text
264
265Returns: nothing
266*/
267
268static void
269native_sha1_mid(sha1 *base, const uschar *text)
270{
271int i;
272uint A, B, C, D, E;
273uint W[80];
274
275base->length += 64;
276
277for (i = 0; i < 16; i++)
278 {
279 W[i] = ((uint)text[0] << 24) | (text[1] << 16) | (text[2] << 8) | text[3];
280 text += 4;
281 }
282
283for (i = 16; i < 80; i++)
284 {
285 register unsigned int x = W[i-3] ^ W[i-8] ^ W[i-14] ^ W[i-16];
286 W[i] = (x << 1) | (x >> 31);
287 }
288
289A = base->H[0];
290B = base->H[1];
291C = base->H[2];
292D = base->H[3];
293E = base->H[4];
294
295for (i = 0; i < 20; i++)
296 {
297 unsigned int T;
298 T = ((A << 5) | (A >> 27)) + ((B & C) | ((~B) & D)) + E + W[i] + 0x5a827999;
299 E = D;
300 D = C;
301 C = (B << 30) | (B >> 2);
302 B = A;
303 A = T;
304 }
305
306for (i = 20; i < 40; i++)
307 {
308 unsigned int T;
309 T = ((A << 5) | (A >> 27)) + (B ^ C ^ D) + E + W[i] + 0x6ed9eba1;
310 E = D;
311 D = C;
312 C = (B << 30) | (B >> 2);
313 B = A;
314 A = T;
315 }
316
317for (i = 40; i < 60; i++)
318 {
319 unsigned int T;
320 T = ((A << 5) | (A >> 27)) + ((B & C) | (B & D) | (C & D)) + E + W[i] +
321 0x8f1bbcdc;
322 E = D;
323 D = C;
324 C = (B << 30) | (B >> 2);
325 B = A;
326 A = T;
327 }
328
329for (i = 60; i < 80; i++)
330 {
331 unsigned int T;
332 T = ((A << 5) | (A >> 27)) + (B ^ C ^ D) + E + W[i] + 0xca62c1d6;
333 E = D;
334 D = C;
335 C = (B << 30) | (B >> 2);
336 B = A;
337 A = T;
338 }
339
340base->H[0] += A;
341base->H[1] += B;
342base->H[2] += C;
343base->H[3] += D;
344base->H[4] += E;
345}

350/*************************************************
351* Process the final text string *
352*************************************************/
353
354/* The string may be of any length. It is padded out according to the rules
355for computing SHA-1 digests. The final result is then converted to text form
356and returned.
357
358Arguments:
359 base pointer to the sha1 storage structure
360 text pointer to the final text vector
361 length length of the final text vector
362 digest points to 16 bytes in which to place the result
363
364Returns: nothing
365*/
366
367static void
368native_sha1_end(sha1 *base, const uschar *text, int length, uschar *digest)
369{
370int i;
371uschar work[64];
372
373/* Process in chunks of 64 until we have less than 64 bytes left. */
374
375while (length >= 64)
376 {
377 native_sha1_mid(base, text);
378 text += 64;
379 length -= 64;
380 }
381
382/* If the remaining string contains more than 55 bytes, we must pad it
383out to 64, process it, and then set up the final chunk as 56 bytes of
384padding. If it has less than 56 bytes, we pad it out to 56 bytes as the
385final chunk. */
386
387memcpy(work, text, length);
388work[length] = 0x80;
389
390if (length > 55)
391 {
392 memset(work+length+1, 0, 63-length);
393 native_sha1_mid(base, work);
394 base->length -= 64;
395 memset(work, 0, 56);
396 }
397else
398 {
399 memset(work+length+1, 0, 55-length);
400 }
401
402/* The final 8 bytes of the final chunk are a 64-bit representation of the
403length of the input string *bits*, before padding, high order word first, and
404high order bytes first in each word. This implementation is designed for short
405strings, and so operates with a single int counter only. */
406
407length += base->length; /* Total length in bytes */
408length <<= 3; /* Total length in bits */
409
410work[63] = length & 0xff;
411work[62] = (length >> 8) & 0xff;
412work[61] = (length >> 16) & 0xff;
413work[60] = (length >> 24) & 0xff;
414
415memset(work+56, 0, 4);
416
417/* Process the final 64-byte chunk */
418
419native_sha1_mid(base, work);
420
421/* Pass back the result, high-order byte first in each word. */
422
423for (i = 0; i < 5; i++)
424 {
425 register int x = base->H[i];
426 *digest++ = (x >> 24) & 0xff;
427 *digest++ = (x >> 16) & 0xff;
428 *digest++ = (x >> 8) & 0xff;
429 *digest++ = x & 0xff;
430 }
431}
