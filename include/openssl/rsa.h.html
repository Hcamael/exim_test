54/*
55 * This flag means the private key operations will be handled by rsa_mod_exp
56 * and that they do not depend on the private key components being present:
57 * for example a key stored in external hardware. Without this flag
58 * bn_mod_exp gets called when private key components are absent.
59 */

62/*
63 * new with 0.9.6j and 0.9.7b; the built-in
64 * RSA implementation now uses blinding by
65 * default (ignoring RSA_FLAG_BLINDING),
66 * but other engines might not need it
67 */

70/*
71 * Does nothing. Previously this switched off constant time behaviour.

73# define RSA_FLAG_NO_CONSTTIME 0x0000

76/* deprecated name for the flag*/
77/*
78 * new with 0.9.7h; the built-in RSA
79 * implementation now uses constant time
80 * modular exponentiation for secret exponents
81 * by default. This flag causes the
82 * faster variable sliding window method to
83 * be used for all exponents.
84 */
