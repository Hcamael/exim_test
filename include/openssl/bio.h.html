203/*
204 * The next three are used in conjunction with the BIO_should_io_special()
205 * condition. After this returns true, BIO *BIO_get_retry_BIO(BIO *bio, int

207 * and the offending BIO. Given a BIO, BIO_get_retry_reason(bio) will return

227/*
228 * The callback is called before and after the underling operation, The
229 * BIO_CB_RETURN flag indicates if it is after the call
230 */
231# define BIO_CB_RETURN 0x80
232# define BIO_CB_return(a) ((a)|BIO_CB_RETURN)
233# define BIO_cb_pre(a) (!((a)&BIO_CB_RETURN))
234# define BIO_cb_post(a) ((a)&BIO_CB_RETURN)
