30/*-
31SSL_CTX -> X509_STORE
32 -> X509_LOOKUP
33 ->X509_LOOKUP_METHOD
34 -> X509_LOOKUP
35 ->X509_LOOKUP_METHOD
36
37SSL -> X509_STORE_CTX
38 ->X509_STORE
39
40The X509_STORE holds the tables etc for verification stuff.
41A X509_STORE_CTX is used while validating a single certificate.
42The X509_STORE has X509_LOOKUPs for looking up certs.
43The X509_STORE then calls a function to actually verify the
44certificate chain.
45*/
46

202# define X509_V_FLAG_POLICY_CHECK 0x80

206# define X509_V_FLAG_INHIBIT_ANY 0x200
207/* Policy variable inhibit-policy-mapping */
208# define X509_V_FLAG_INHIBIT_MAP 0x400
209/* Notify callback that policy is OK */

212# define X509_V_FLAG_EXTENDED_CRL_SUPPORT 0x1000
213/* Delta CRL support */
214# define X509_V_FLAG_USE_DELTAS 0x2000
215/* Check self-signed CA signature */
216# define X509_V_FLAG_CHECK_SS_SIGNATURE 0x4000
217/* Use trusted store first */
218# define X509_V_FLAG_TRUSTED_FIRST 0x8000

220# define X509_V_FLAG_SUITEB_128_LOS_ONLY 0x10000
221/* Suite B 192 bit only mode */
222# define X509_V_FLAG_SUITEB_192_LOS 0x20000
223/* Suite B 128 bit mode allowing 192 bit algorithms */
224# define X509_V_FLAG_SUITEB_128_LOS 0x30000
